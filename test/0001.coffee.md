    should = require 'should'
    path = require 'path'
    FS = require '../src/esl'
    async = require 'async'

    client_port = 8024
    server_port = 8022

    describe 'The client', ->
      describe 'when it is started', ->

        it 'should be reachable', (done) ->
          client = FS.client ->
            client.end()
            done()
          .connect client_port, '127.0.0.1'

        it 'should accept the connection', (done) ->
          client = FS.client ->
            @headers['Reply-Text'].should.eql '+OK accepted'
            client.end()
            done()
          .connect client_port, '127.0.0.1'

        it 'should reloadxml', (done) ->
          cmd = 'reloadxml'
          client = FS.client ->
            # @trace on
            # @debug on
            @api cmd
            .then ->
              @body.should.match /\+OK \[Success\]/
              this
            .then -> @exit()
            .then -> client.end()
            .then done
          .connect client_port, '127.0.0.1'
